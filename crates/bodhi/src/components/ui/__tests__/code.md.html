import { useState } from 'react';

// This line will be highlighted
function Counter() {
  // These lines will be highlighted
  const [count, setCount] = useState(0);
  const increment = () => setCount(c => c + 1);
  
  return (
    <button onClick={increment}>
      Count: {count}
    </button>
  );
}
